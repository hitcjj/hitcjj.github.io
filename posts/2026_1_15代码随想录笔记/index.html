##题目203.移除链表元素

注意点：如果头节点可能被删除，那么要在头节点之前添加一个哨兵节点，这样我们无需特判头节点被删除的情况

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {
        if(!head)
        return nullptr;
        auto new_head=new ListNode(0,head);
        auto p=new_head;
        while(p->next)
        {
            if(p->next->val==val)
            {
                p->next=p->next->next;
            }
            else{
                p=p->next;
            }
        }
        return new_head->next;
    }
};

#题目 707.设计链表 #边界条件很容易错误


 typedef struct MyListNode{
        int val;
        struct MyListNode* next;
    }MyListNode;
class MyLinkedList {

public:
    MyListNode* head;
    MyLinkedList() {
        head=new MyListNode(-1,nullptr);
    }
    
    int get(int index) {
         if(index>=head->val+1)
        return -1;
        auto p=head;
        for(int i=-1;i<index;i++)
        {
            p=p->next;
        }
        return p->val;
    }
    
    void addAtHead(int val) {
        MyListNode*p=new MyListNode(val,head->next);
        head->next=p;
        head->val++;
    }
    
    void addAtTail(int val) {
        auto p=head;
        MyListNode*k=new MyListNode(val,nullptr);
        while(p->next)
        {
            p=p->next;
        }
        p->next=k;
        head->val++;
    }
    
    void addAtIndex(int index, int val) {
        if(index>head->val+1)
        return ;
        auto p=head;
        for(int i=0;i<index;i++)
        {
            p=p->next;
        }
        MyListNode*k=new MyListNode(val,p->next);
        p->next=k;
        head->val++;
    }
    
    void deleteAtIndex(int index) {
        if(index>head->val||head->val==-1)
        return ;
        auto p=head;
        for(int i=0;p&&i<index;i++)
        {
            p=p->next;
        }
        auto x=p->next;
        p->next=x->next;
        delete x;
        head->val--;
    }
};

#题目 206.反转链表 #使用头插法，不难

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(!head)
        return nullptr;
        ListNode *p=head;
        ListNode *new_node=new ListNode(0,nullptr);
        while(p)
        {
            auto q=p;
            p=p->next;
            q->next=new_node->next;
            new_node->next=q;
        }
        return new_node->next;

    }
};
