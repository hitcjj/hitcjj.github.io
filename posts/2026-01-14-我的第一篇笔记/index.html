##题目 704. 二分查找

#注意点：左闭右闭的理解：左右都属于有效访问，所以left<=right需要等号，right=mid-1 左闭右开,右边无效，因此不用等号，right=mid

解题代码（c++）#

def binary_search(nums, target):
    """
    二分查找实现：有序数组中查找目标值
    :param nums: 升序排列的整型数组
    :param target: 要查找的目标值
    :return: 目标值下标，不存在则返回-1
    """
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = (left + right) // 2  # 避免溢出也可以写 left + (right - left) // 2
        if nums[mid] == target:
            return mid
        elif nums[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

##题目27.移除元素

#注意不要死循环

int removeElement(vector<int>& nums, int val) {
        int n=nums.size();
        int i=0;
        while(i<n)
        {
            if(nums[i]==val)
            {
                n--;
                for(int j=i;j<nums.size()-1;j++)
                {
                    nums[j]=nums[j+1];
                }
            }
            else{
                i++;
            }
        }
        return n;
